def log_loss(loss, n, log_every):
  if n % log_every == 0:
    print(f'Epoch {n}:')
    print(f'\tLoss = {loss[0].item()}')
    for key, val in loss[1].items():
      if key == 'props' or key == 'dprops':
        print(f'\t{key} = {val}')
      else:
        print(f'\t{key} = {val.item()}')
